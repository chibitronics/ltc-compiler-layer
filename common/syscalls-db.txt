# Database of syscalls.
# Lines beginning with a # are ignored.
# Fields are | delimited
# fields:
# Valid types:
#    c - Standard C function
#    x - Standard C++ function
#    C - C++ constructor
#    D - C++ destructor
#    m - C++ member function
#
# c fields (standard C function):
# type | name
#
# x fields (standard C++ function):
# type | name | return type | arguments | names
# e.g. int digitalRead(int pin) -> "x | digitalRead | int | int pin | pin"
#
# The following types are kept around for historical reasons, but due to
# the uniquely awful nature of C++, they cannot be used.
#
# C fields (C++ constructor):
# type | class | arguments (with variable names) | comma list of variable names
# e.g.  Print::Print(int a, int b); -> "C | Print | int a, int b | a, b"
#
# D fields (C++ destructor):
# type | class | arguments (with variable names) | comma list of variable names
# e.g.  Print::~Print(int a, int b); -> "D | Print | int a, int b | a, b"
#
# m fields (C++ member function):
# type | class | method | return type | arguments | names
